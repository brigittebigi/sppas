310      @return: boolean 
311   
312      """ 
313      functions = name.split("_") 
314      module = globals() 
315   
316      if len(functions) == 1: 
317      # Only one function of this module will be used. 
318   
319          # get the function corresponding to the given name 
320          func = module.get(functions[0], None) 
321          if not func: 
322              raise Exception('Unknown function: %s' % name) 
323   
324          # specific case of string/unicode labels, except if regexp 
325          if "regexp" not in name and isinstance(arg,str): 
326              arg = " ".join(arg.split()) 
327   
328          if opt=="any": 
329              def wrap(annotation): 
330                  return any(func(l.GetTypedValue(), arg) for l in annotation.GetLabel().GetLabels()) 

331          else: 
332              def wrap(annotation): 
333                  return func(annotation.GetLabel().GetTypedValue(), arg) 
334   
335      elif len(functions) == 2: 
336      # Two functions of this module will be used. 
337   
338          func = module.get(functions[0], None) 
339          comparator = getattr(operator, functions[1], None) 
340          if None in (func, comparator): 
341              raise Exception('Unknown function: %s' % name) 
342   
343          def wrap(annotation): 
344              time = func(annotation) 
345              return comparator(time, arg) 
346   
347      else: 
348      # No more functions are accepted! 
349          raise Exception('Unknown function: %s' % name) 
350
